#### 1. 创建 Statistics 类

**文件**: `videoDecoder/Statistics.h/cpp`

```cpp
class Statistics {
public:
    struct VideoStats {
        int decodedFrames = 0;      // 解码帧数
        int renderedFrames = 0;     // 渲染帧数
        int droppedFrames = 0;      // 丢帧数
        double fps = 0.0;           // 当前帧率
        int bitrate = 0;            // 码率
        double decodeTime = 0.0;   // 解码时间（ms）
        double renderTime = 0.0;   // 渲染时间（ms）
    };
    
    struct AudioStats {
        int decodedFrames = 0;      // 解码帧数
        int playedFrames = 0;       // 播放帧数
        int bitrate = 0;            // 码率
        double decodeTime = 0.0;   // 解码时间（ms）
    };
    
    struct SyncStats {
        double videoClock = 0.0;     // 视频时钟
        double audioClock = 0.0;    // 音频时钟
        double syncDiff = 0.0;      // 同步差异（ms）
        int syncErrors = 0;         // 同步错误次数
    };
    
    // 更新统计
    void updateVideoStats(const VideoStats& stats);
    void updateAudioStats(const AudioStats& stats);
    void updateSyncStats(const SyncStats& stats);
    
    // 获取统计
    VideoStats videoStats() const { return videoStats_; }
    AudioStats audioStats() const { return audioStats_; }
    SyncStats syncStats() const { return syncStats_; }
    
    // 重置统计
    void reset();
    
signals:
    void statsUpdated();  // 统计更新信号

private:
    VideoStats videoStats_;
    AudioStats audioStats_;
    SyncStats syncStats_;
    std::mutex mutex_;
};
```

- **AudioThread**: 更新解码帧数、播放帧数

class Playlist : public QObject {
    Q_OBJECT
public:
    // 添加/删除项目
    void addItem(const PlaylistItem& item);
    void removeItem(int index);
    void clear();
    
    // 播放控制
    void playNext();
    void playPrevious();
    void playAt(int index);
    
    // 播放模式
    enum PlayMode {
        Sequential,    // 顺序播放
        Loop,          // 循环播放
        Random,        // 随机播放
        SingleLoop     // 单曲循环
    };
    void setPlayMode(PlayMode mode);
    
    // 获取当前项目
    PlaylistItem currentItem() const;
    int currentIndex() const;
    
signals:
    void currentItemChanged(int index);
    void itemAdded(int index);
    void itemRemoved(int index);
    void playModeChanged(PlayMode mode);

private:
    QList<PlaylistItem> items_;
    int currentIndex_ = -1;
    PlayMode playMode_ = Sequential;
};
